justice.js

github.com/okor/justice | @okor

How to use

  <script type="text/javascript" src="justice.min.js"></script>
  <script type="text/javascript">
    Justice.init();
  </script>

With options

  <script type="text/javascript" src="justice.min.js"></script>
  <script type="text/javascript">
    Justice.init({
      metrics: {
        TTFB:             { budget: 200   },
        domInteractive:   { budget: 250   },
        domComplete:      { budget: 800   },
        firstPaint:       { budget: 1000  },
        pageLoad:         { budget: 2000  },
        requests:         { budget: 6     },
      },
      warnThreshold: 0.08,
      showFPS: true,
      chartType: 'spline'
    });
  </script>
